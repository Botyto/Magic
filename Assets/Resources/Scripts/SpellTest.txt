Class.SpellTest = {
	__inherit = "InstantSpell",
}

function SpellTest:Cast()
	local handle = self:ManifestEnergyAndFocus(100, Vector3.new(0,0,5), Energy.Element.Fire, Energy.Shape.Cube)
	self:OrientTowards(handle, self:GetTargetPosition());
	local forceDirection = Vector3.new(0, 0, 1) --self:GetTargetPosition() - self:GetFocusPosition(handle);
	self:ApplyForce(handle, forceDirection * (5 * self:GetParam().level), ForceMode.Impulse)
end
